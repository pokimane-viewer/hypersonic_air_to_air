# CAE/FEA (often paired with CAD) solves large sparse linear systems Ax = b for physics simulations.
import numpy as np

# demo: small linear system
A = np.array([[3., 2.],
              [1., 2.]])
b = np.array([5., 5.])
x = np.linalg.solve(A, b)
print(x)  # [1. 2.]

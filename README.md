# hypersonic_air_to_air and demo solve small linear system with numpy


import numpy as np


A = np.array([[3., 2.],
              [1., 2.]])
b = np.array([5., 5.])
x = np.linalg.solve(A, b)
print(x)  # [1. 2.]
